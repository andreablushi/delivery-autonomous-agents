\section{BDI Agent}

% ----- Beliefs: Tracker ----------------------------------------
\begin{frame}{Beliefs: Tracker}
  \begin{itemize}
    \item Stores latest observation $\langle v,\, t_{seen} \rangle$ per entity
    \item Updated on each sensing report
    \item Invalidated when entity is visible-but-absent
  \end{itemize}
  \begin{block}{Confidence decay}
    \[ c(t) = \exp\!\left(-\frac{t - t_{seen}}{\tau}\right) \]
  \end{block}
\end{frame}

% ----- Beliefs: Memory -----------------------------------------
\begin{frame}{Beliefs: Memory}
  \begin{itemize}
    \item Bounded temporal history of observations per entity
    \item Entries older than a configurable TTL are evicted automatically
    \item Tracks enemy position history for heat-field computation
  \end{itemize}
\end{frame}

% ----- Beliefs: Agents -----------------------------------------
\begin{frame}{Beliefs: Agents}
  \begin{itemize}
    \item \textbf{Self} — own position and current state
    \item \textbf{Friends} — latest state per teammate via Tracker
    \item \textbf{Enemies} — latest state via Tracker; position history via Memory
    \item \textbf{EnemyPredictor} — estimates each enemy's next position from history
    \item Spatial heat field — aggregates past enemy observations across the map
  \end{itemize}
\end{frame}

% ----- Beliefs: Map & Crates --------------------------------------------
\begin{frame}{Beliefs: Map \& Crates}
  \begin{itemize}
    \item Static tile layout, transient blocked tiles, traversal penalties
    \item Crates tracked as dynamic obstacles; latest position via Tracker;
  \end{itemize}
\end{frame}

% ----- Beliefs: Map --------------------------------------------
\begin{frame}{Beliefs: Map --- Safe Zones}
  Directional Tiles create one-way edges — may trap agents in dead zones
  \begin{block}{Safe SCC (Tarjan's algorithm)}
    \begin{itemize}
        \item Map is represented as a directed graph with edges for valid movements;
        \item Tarjan's algorithm identifies SCCs in this graph;
        \item An SCC is \emph{safe} iff it contains $\geq 1$ spawn tile \textbf{and} $\geq 1$ delivery tile;
        \item All tiles outside safe SCCs are reclassified as walls.
    \end{itemize}

  \end{block}
\end{frame}

% ----- Beliefs: Parcels ----------------------------------------
\begin{frame}{Beliefs: Parcels}
  \begin{itemize}
    \item Reward decay is applied even when out of sensor range
    \item Parcel $p$ removed from beliefs when $r_p(t) \leq 0$
  \end{itemize}
  \begin{block}{Estimated reward}
    \[ r_p(t) = r_{p,0} - \left\lfloor \frac{t - t_0}{\Delta_t} \right\rfloor \]
  \end{block}
\end{frame}

% ----- Desire: Distance Metric ---------------------------------
\begin{frame}{Desire: Distance Metric}
  \begin{itemize}
    \item A* search with dynamic tile costs based on beliefs
    \item Walls and blocked tiles are impassable; crates are high-cost
    \item Distance $d$ is the total cost of the A* path to the target
  \end{itemize}
\end{frame}

% ----- Desire: ReachParcel -------------------------------------
\begin{frame}{Desire: ReachParcel}
  \begin{itemize}
    \item Navigate to and collect a visible parcel
    \item Priority 1 — competes directly with DeliverParcel
  \end{itemize}
  \begin{block}{Score}
    \[ S = \frac{r_p + \sum_{c \in C} r_c}{d} \cdot \gamma, \qquad
       \gamma = \min\!\left(1,\, \frac{d_{enemy}}{d}\right) \]
    \begin{itemize}
        \item $r_p$: parcel reward;
        \item $\sum_{c \in C} r_c$: value of carried stack;
        \item $\gamma$: race factor — penalizes parcels where enemies have advantage;
    \end{itemize}
  \end{block}
\end{frame}

% ----- Desire: DeliverParcel -----------------------------------
\begin{frame}{Desire: DeliverParcel}
  \begin{itemize}
    \item Navigate to delivery tile while carrying parcels
    \item Priority 1 — competes directly with ReachParcel
  \end{itemize}
  \begin{block}{Score}
    \[ S = \frac{\sum_{c \in C} r_c}{d} \]
  \end{block}
\end{frame}

% ----- Desire: Explore -----------------------------------------
\begin{frame}{Desire: Explore}
  \begin{itemize}
    \item Seek unvisited spawn tile when no parcels are reachable
    \item Priority 0 — fallback;
    \item $w$: cluster density;\quad $\tau$: freshness;\quad $h$: enemy heat
  \end{itemize}
  \begin{block}{Score}
    \[ S = \frac{w \cdot \tau}{d \cdot (1 + h)} \]
  \end{block}
\end{frame}

% ----- Desire: Explore — Variables ----------------------------
\begin{frame}{Desire: Explore --- Variables}
  \begin{block}{Freshness $\tau$}
    \[ \tau = \min\!\left(1,\, \frac{\Delta t}{T_{\text{spawn}}}\right) \]
    $\Delta t$: time since spawn tile last observed;\quad $T_{\text{spawn}}$: spawn interval
  \end{block}
  \begin{block}{Cluster weight $w$}
    \[ w = \sum_{y \in OST} \frac{1}{\delta(x,\,y)+1} \]
    $OST$: spawn tiles within observation range $R_{\text{obs}}$
  \end{block}
  \begin{block}{Enemy heat $h$}
    \[ h = \sum_{e \in E}
         \exp\!\left(-\frac{d_e^2}{2\sigma^2}\right)
         \exp\!\left(-\frac{\Delta t_e}{\tau_h}\right) \]
    Spatial $\times$ temporal decay over all tracked enemy observations
  \end{block}
\end{frame}

% ----- Collision: Detection ------------------------------------
\begin{frame}{Collision: Detection}
  Checked before every move step. Target tile is \emph{occupied} if:
  \begin{itemize}
    \item An enemy is currently observed on it, \textbf{or}
    \item An enemy's \emph{predicted} next position matches it with sufficient confidence
  \end{itemize}
  \begin{block}{Enemy position prediction}
    Next position estimated from last known state via \texttt{EnemyPredictor};
    only predictions above a confidence threshold trigger a block.
  \end{block}
\end{frame}

% ----- Collision: Escalation -----------------------------------
\begin{frame}{Collision: Escalation}
  \begin{columns}
    \column{0.58\textwidth}
      \begin{enumerate}
        \item \textbf{Detour} — replan with tile as wall;\\
              accepted only if extra steps $\leq \Delta_{th}$
        \item \textbf{Wait} — no viable detour:\\
              pause for random duration $\sim \mathcal{U}$
        \item \textbf{Block \& replan} — after wait expires\\
              \emph{or} after $L$ repeated invalidations:\\
              mark tile blocked for random TTL, full replan
      \end{enumerate}
